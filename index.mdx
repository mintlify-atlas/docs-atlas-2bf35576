  <h2 className="text-3xl font-bold dark:text-white text-gray-900 mb-4">Quick start</h2>
  <p className="text-base dark:text-gray-400 text-gray-600 mb-8">Get up and running with ByteKit in minutes</p>
  
  <Steps>
    <Step title="Install ByteKit">
      Install the package using your preferred package manager:
      
      <CodeGroup>
      ```bash npm
      npm install bytekit
      ```
      ```bash yarn
      yarn add bytekit
      ```
      ```bash pnpm
      pnpm add bytekit
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Import and configure">
      Import the modules you need. ByteKit is fully tree-shakeable:
      
      ```typescript
      import { ApiClient, createLogger, StringUtils } from "bytekit";
      
      const api = new ApiClient({
        baseUrl: "https://api.example.com",
        defaultHeaders: { "X-API-Key": process.env.API_KEY }
      });
      
      const logger = createLogger({ 
        namespace: "app", 
        level: "info" 
      });
      ```
    </Step>
    
    <Step title="Start building">
      Use ByteKit's utilities in your application:
      
      ```typescript
      // Make API calls with automatic retries
      const users = await api.get("/users");
      
      // Log structured messages
      logger.info("Users fetched", { count: users.length });
      
      // Use helper utilities
      const slug = StringUtils.slugify("New Feature – 2024");
      ```
    </Step>
  </Steps>
</div>
